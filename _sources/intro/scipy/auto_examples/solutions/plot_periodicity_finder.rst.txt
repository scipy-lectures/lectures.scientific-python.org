    import scipy as sp

    ft_populations = sp.fft.fft(populations, axis=0)
    frequencies = sp.fft.fftfreq(populations.shape[0], years[1] - years[0])
    periods = 1 / frequencies

    plt.figure()
    plt.plot(periods, abs(ft_populations) * 1e-3, "o")
    plt.xlim(0, 22)
    plt.xlabel("Period")
    plt.ylabel(r"Power ($\cdot10^3$)")

    plt.show()
