    x, y = np.indices((100, 100))
    sig = (
        np.sin(2 * np.pi * x / 50.0)
        * np.sin(2 * np.pi * y / 50.0)
        * (1 + x * y / 50.0**2) ** 2
    )
    mask = sig > 1

    plt.figure(figsize=(7, 3.5))
    plt.subplot(1, 2, 1)
    plt.imshow(sig)
    plt.axis("off")
    plt.title("sig")

    plt.subplot(1, 2, 2)
    plt.imshow(mask, cmap=plt.cm.gray)
    plt.axis("off")
    plt.title("mask")
