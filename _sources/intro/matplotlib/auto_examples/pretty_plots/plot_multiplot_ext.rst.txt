    from matplotlib.patches import FancyBboxPatch

    ax = plt.gca()
    ax.add_patch(
        FancyBboxPatch(

            width=0.66,
            height=0.34,
            clip_on=False,
            boxstyle="square,pad=0",

            transform=plt.gca().transAxes,
