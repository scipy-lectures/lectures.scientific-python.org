    import matplotlib.pyplot as plt

    n = 256
    X = np.linspace(0, 2, n)
    Y = np.sin(2 * np.pi * X)

    plt.plot(X, Y, lw=2, color="violet")
    plt.xlim(-0.2, 2.2)
